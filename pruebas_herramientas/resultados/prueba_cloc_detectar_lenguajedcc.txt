import json
from pathlib import Path
import tkinter as tk
from tkinter import filedialog
import sys
repo_root = Path(__file__).resolve().parents[2]
src_path = repo_root / "src"
sys.path.insert(0, str(src_path))
from analizador_calidad_software.cli import (
    ejecutar_herramienta,
    obtener_ruta_herramienta
)
def elegir_tipo_entrada() -> str:
    print("¿Que quieres analizar?")
    print("1. Un archivo")
    print("2. Un directorio/carpeta")
    opcion = input("Selecciona una opción (1 o 2):").strip()
    if opcion == "1":
        return "archivo"
    elif opcion == "2":
        return "directorio"
    else:
        raise RuntimeError("Opción no válida")
def seleccionadr_ruta(tipo_archivo: str) -> Path:
    root = tk.Tk()
    root.withdraw()
    if tipo_archivo == "archivo":
        ruta = filedialog.askopenfilename(
            title="Selecciona un ficero para analizar con cloc"
        )
    elif tipo_archivo == "directorio":
        ruta = filedialog.askdirectory(
            title="Selecciona un directorio/carpeta para analizar con cloc"
        )
    if not ruta:
        raise RuntimeError("No se seleccionó ninguna ruta")
    return Path(ruta)
def obtener_lenguaje_principal(salida__json: str) -> str:
    datos = json.loads(salida__json)
    lenguajes = []
    for clave, valor in datos.items():
        if clave in ("header", "SUM"):
            continue
        if isinstance(valor, dict) and "code" in valor:
            lenguajes.append((clave, valor["code"]))
    if not lenguajes:
        return "Desconocido"
    lenguajes.sort(key=lambda x: x[1], reverse=True)
    return lenguajes[0][0]
def main() -> None:
    tipo_archivo = elegir_tipo_entrada()
    ruta = seleccionadr_ruta(tipo_archivo)
    if ruta.is_file():
        print("Se ha seleccionado un archivo")
    if ruta.is_dir():
        print("Se ha selecionado un directorio")
    ruta_cloc = obtener_ruta_herramienta("cloc")
    resultado = ejecutar_herramienta([str(ruta_cloc), str(ruta), "--json"])
    if resultado.returncode !=0:
        print("Error al ejecutar cloc")
        print(resultado.stderr)
        return
    lenguaje = obtener_lenguaje_principal(resultado.stdout)
    print(f"Ruta analizada: {ruta}")
    print(f"Lenguaje detectado: {lenguaje}")
if __name__ == "__main__":
    main()
