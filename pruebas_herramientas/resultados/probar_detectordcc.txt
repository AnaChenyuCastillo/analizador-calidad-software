from pathlib import Path
import sys
from tkinter import filedialog
repo_root = Path(__file__).resolve().parents[2]
src_path = repo_root / "src"
sys.path.insert(0, str(src_path))
from analizador_calidad_software.deteccion.detector_lenguaje import (
    detectar_lenguaje_proyecto,
    obtener_ranking_lenguajes_proyecto,
)
def main() -> None:
    ruta_texto = input("Introduce la ruta del proyecto a analizar: ").strip()
    ruta = filedialog.askdirectory(
            title="Selecciona un directorio/carpeta para analizar con cloc"
        )
    ruta_proyecto = Path(ruta).resolve()
    if not ruta_proyecto.exists():
        print("Error: la ruta no existe.")
        return
    if not ruta_proyecto.is_dir():
        print("Error: la ruta no corresponde a una carpeta.")
        return
    ranking = obtener_ranking_lenguajes_proyecto(ruta_proyecto)
    lenguaje = detectar_lenguaje_proyecto(ruta_proyecto)
    print("\n=== RESULTADO DEL MÓDULO DE DETECCIÓN ===")
    print(f"Ruta analizada: {ruta_proyecto}")
    print(f"Lenguaje detectado: {lenguaje}")
    print(f"Ranking de lenguajes: {ranking}")
if __name__ == "__main__":
    main()
