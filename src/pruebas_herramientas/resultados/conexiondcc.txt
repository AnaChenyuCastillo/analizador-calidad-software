package acceso_a_datos ;
import java.sql.Connection ;
import java.sql.DriverManager ;
import java.sql.SQLException ;
import java.sql.Statement ;
class conexion public class conexion
{
    private String mensaje ;
    private Connection dbConnection ;
    Statement stm ;
    private boolean conexionexitosa ;
    private String Usuario ;
    private String clave ;
    metodo conexion public conexion ()
    {
        this.mensaje = "" ;
        this.conexionexitosa = false ;
    }
    metodo establecerconexion public String establecerconexion (String Usuario,String Clave)
    {
        this.mensaje = "conexion correcta" ;
        try
        {
            Class.forName ("com.mysql.jdbc.Driver") ;
            String url="jdbc:mysql://localhost:3306/bd_libreria" ;
            this.dbConnection = DriverManager.getConnection (url, Usuario, Clave) ;
            this.stm= dbConnection.createStatement () ;
            this.conexionexitosa = true ;
            this.Usuario = Usuario ;
            this.clave = Clave ;
        }
        catch (Exception e)
        {
            this.mensaje = "error en conexion " ;
            this.conexionexitosa = false ;
        }
        return this.mensaje ;
    }
    metodo getconexioncorrecta public boolean getconexioncorrecta ()
    {
        return this.conexionexitosa ;
    }
    metodo cerrarConexion public String cerrarConexion ()
    {
        this.mensaje = "desconexion correcta" ;
        try
        {
            this.stm.close () ;
            this.dbConnection.close () ;
        }
        catch (Exception e)
        {
            this.mensaje = "error en desconexion" + e.toString () ;
        }
        return this.mensaje ;
    }
    metodo getConexion public Connection getConexion ()
    {
        return this.dbConnection ;
    }
    metodo getEstado public Statement getEstado ()
    {
        return this.stm ;
    }
    metodo getUsuario public String getUsuario ()
    {
        return this.Usuario ;
    }
    metodo getClave public String getClave ()
    {
        return this.clave ;
    }
}
